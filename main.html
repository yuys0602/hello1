const express = require('express');
const WebSocket = require('ws');
const { Pool } = require('pg');

const app = express();
const pool = new Pool({
  connectionString: process.env.DATABASE_URL
});

// RESTful API - 获取生产数据
app.get('/api/production/:lineId', async (req, res) => {
  try {
    const { lineId } = req.params;
    const { startDate, endDate } = req.query;
    
    const query = `
      SELECT 
        line_id,
        DATE_TRUNC('hour', timestamp) as hour,
        SUM(output) as total_output,
        AVG(quality_rate) as avg_quality,
        AVG(oee) as avg_oee
      FROM production_records
      WHERE line_id = $1 
        AND timestamp BETWEEN $2 AND $3
      GROUP BY line_id, hour
      ORDER BY hour DESC
    `;
    
    const result = await pool.query(query, [lineId, startDate, endDate]);
    res.json({
      success: true,
      data: result.rows
    });
  } catch (error) {
    res.status(500).json({ 
      success: false, 
      error: error.message 
    });
  }
});

AI生成的算法代码示例

import pandas as pd
import numpy as np
from scipy import stats
from sklearn.ensemble import RandomForestRegressor

class ProductionAnalyzer:
    def calculate_oee(self, production_data):
        """计算OEE（设备综合效率）"""
        availability = production_data['run_time'] / production_data['planned_time']
        performance = production_data['actual_output'] / production_data['theoretical_output']
        quality = production_data['good_products'] / production_data['total_products']
        
        oee = availability * performance * quality
        return {
            'oee': round(oee * 100, 2),
            'availability': round(availability * 100, 2),
            'performance': round(performance * 100, 2),
            'quality': round(quality * 100, 2)
        }
    
    def detect_anomalies(self, time_series_data):
        """使用Z-score方法检测生产异常"""
        z_scores = np.abs(stats.zscore(time_series_data))
        threshold = 3
        anomalies = time_series_data[z_scores > threshold]
        
        return {
            'anomaly_indices': np.where(z_scores > threshold)[0].tolist(),
            'anomaly_values': anomalies.tolist()
        }
    
    def predict_production(self, historical_data):
        """预测未来24小时产量"""
        # 特征工程
        features = ['hour', 'day_of_week', 'shift', 'temperature', 'humidity']
        X = historical_data[features]
        y = historical_data['output']
        
        # 训练模型
        model = RandomForestRegressor(n_estimators=100)
        model.fit(X, y)
        
        # 预测未来24小时
        future_features = self.generate_future_features(24)
        predictions = model.predict(future_features)
        
        return predictions
